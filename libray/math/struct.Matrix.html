pub fn transpose(&self) -> Matrix

Transposes Matrix and returns a new Matrix

Example
use libray::math::Matrix;

let a = vec![
    vec![0.0, 9.0, 3.0, 0.0],
    vec![9.0, 8.0, 0.0, 8.0],
    vec![1.0, 8.0, 5.0, 3.0],
    vec![0.0, 0.0, 5.0, 8.0],
];
let a = Matrix::from(a);

let reference = vec![
    vec![0.0, 9.0, 1.0, 0.0],
    vec![9.0, 8.0, 8.0, 0.0],
    vec![3.0, 0.0, 5.0, 5.0],
    vec![0.0, 8.0, 3.0, 8.0],
];
let reference = Matrix::from(reference);

assert_eq!(reference, a.transpose());
